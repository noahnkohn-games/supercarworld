// --- LÓGICA DO HOST --- function createRoom() { myName = document.getElementById('player-name').value || "Host"; setStatus("Criando sala, aguarde..."); console.log("[HOST] Tentando conectar ao servidor PeerJS para criar sala..."); peer = new Peer(); // NOVO: Sistema de auto-reconexão peer.on('disconnected', function() { console.warn("[HOST] Desconectado do servidor PeerJS. Tentando reconectar automaticamente..."); peer.reconnect(); }); peer.on('open', function(id) { myId = id; isHost = true; console.log("[HOST] Sala criada com sucesso! ID:", id); startGameEngine(); }); peer.on('connection', function(conn) { console.log(`[HOST] Nova tentativa de conexão recebida do ID: ${conn.peer}`); clientConnections.push(conn); conn.on('open', function() { console.log(`[HOST] Conexão ESTABELECIDA com o cliente: ${conn.peer}`); }); conn.on('data', function(data) { if(data.type === 'join') { console.log(`[HOST] Dados iniciais (join) recebidos do cliente ${data.name} (${data.id})`); players[data.id] = { name: data.name, color: data.color, x: 0, z: 0, rotation: 0, mesh: createCarMesh(data.color) }; } else if (data.type === 'update') { if(players[data.id]) { players[data.id].x = data.x; players[data.id].z = data.z; players[data.id].rotation = data.rotation; } } }); conn.on('close', function() { console.log(`[HOST] Cliente ${conn.peer} desconectou-se.`); clientConnections = clientConnections.filter(c => c.peer !== conn.peer); if(players[conn.peer] && players[conn.peer].mesh) { scene.remove(players[conn.peer].mesh); } delete players[conn.peer]; }); conn.on('error', function(err) { console.error(`[HOST] Erro na conexão com o cliente ${conn.peer}:`, err); }); }); peer.on('error', function(err) { console.error("[HOST] Erro no servidor PeerJS:", err); setStatus("Erro ao criar sala: " + err.type); }); } // --- LÓGICA DO CLIENTE (CONVIDADO) --- function joinRoom() { const hostId = document.getElementById('room-id').value.trim(); if(!hostId) return setStatus("Digite o ID da sala primeiro!"); myName = document.getElementById('player-name').value || "Convidado"; setStatus("Conectando ao Host... Aguarde."); console.log(`[CLIENTE] Tentando conectar à sala ID: ${hostId}`); peer = new Peer(); // NOVO: Sistema de auto-reconexão peer.on('disconnected', function() { console.warn("[CLIENTE] Desconectado do servidor PeerJS. Tentando reconectar automaticamente..."); peer.reconnect(); }); peer.on('open', function(id) { myId = id; console.log(`[CLIENTE] Conectado ao servidor PeerJS. Meu ID: ${myId}`); console.log(`[CLIENTE] Solicitando conexão direta com o Host: ${hostId}`); // reliable: true força o WebRTC a tentar garantir a entrega dos dados de conexão hostConnection = peer.connect(hostId, { reliable: true }); hostConnection.on('open', function() { console.log("[CLIENTE] Conexão direta com o Host ABERTA!"); setStatus("Conectado! Entrando no mundo..."); hostConnection.send({ type: 'join', id: myId, name: myName, color: myColor }); startGameEngine(); }); hostConnection.on('data', function(state) { if(state.type === 'stateUpdate') { syncWorldState(state.playersData); } }); hostConnection.on('close', function() { console.warn("[CLIENTE] A conexão com o Host foi fechada."); alert("O Host fechou a sala ou a conexão caiu!"); window.location.reload(); }); hostConnection.on('error', function(err) { console.error("[CLIENTE] Erro na conexão com o Host:", err); setStatus("Erro ao conectar com o Host."); }); }); peer.on('error', function(err) { console.error("[CLIENTE] Erro no servidor PeerJS:", err); if (err.type === 'peer-unavailable') { setStatus("Erro: Sala não encontrada. O Host fechou ou o ID está errado."); } else if (err.type === 'network' || err.type === 'server-error') { // Se for erro de rede, o evento 'disconnected' acima tentará reconectar console.warn("[CLIENTE] Falha na rede, o auto-reconectar deve assumir."); } else { setStatus("Erro de rede: " + err.type); } }); }
